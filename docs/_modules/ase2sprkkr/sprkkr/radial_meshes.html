[docs]  @staticmethod
  def interpolator_for_coors(coors, values):
      v1 = abs(values[0])
      v2 = abs(values[-1])

      if v1 == 0.:
          r_method = False
      elif v2 == 0.:
          r_method = True
      else:
          c1 = coors[0]
          c2 = coors[-1]
          d = (v1 - v2)
          dr = (v1 * c1 - v2 * c2)
          r_method = abs(d) > abs(dr)

      if r_method:
          spl = CubicSpline(coors, values * coors, extrapolate=True)
          return lambda x: spl(x) / x

      return CubicSpline(coors, values)




[docs]def _clearing_property(name):

    attr = '_' + name

    def reader(self):
        return getattr(self, attr)

    def writer(self, value):
        setattr(self, attr, value)
        self._clear()

    out = property(reader)
    out = out.setter(writer)
    return out

[docs]  def _clear(self):
      self._coors = None


  @property
  def coors(self):
      if self._coors is None:
          self._coors = np.empty(self.jrws)
          self._coors[0] = r = self.r1
          dx = math.exp(self.dx)
          for i in range(1, self.jrws):
              self._coors[i] = r = r * dx
      return self._coors
