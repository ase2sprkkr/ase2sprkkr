[docs]def get_sprkkr_input(task):

    lines='None'

    if task.upper() == 'ARPES':
        lines=dict_inp_data['ARPES']
    elif task.upper() == 'SCF':
        lines=dict_inp_data['SCF']
    elif task.upper() == 'DOS':
        lines=dict_inp_data['DOS']
    elif task.upper() == 'PHAGEN':
        lines=dict_inp_data['PHAGEN']
    else:
        lines='None'
    return  str2lines(lines)
